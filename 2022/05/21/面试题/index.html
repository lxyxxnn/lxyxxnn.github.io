第一题:Object与Map的区别

Map是Object的一个子类
可以有序保存任意类型的数据
使用键值存储，其中键可以存储任意类型const m = new Map()可以得到一个map实例

map:通过size

object： 没有实现迭代器，需要自行实现，不实现只能for-in循环迭代，顺序不确定

ES5实现四种方式

1.原型链继承
====直接让子类的原型对象指向父类实例，当子类实例找不到对应
====的属性和方法就向他的原型对象，也就是父实例上找
function Person() {

}
function Student() {

缺点，修改来自父类的引用类型变量会影响所有Student实例

2.构造函数的继承
在子类中执行父类的构造函数，并为其绑定子类的this，让 父类构造函数
把成员属性和方法都挂在子类上
function Persion(name) {
this.name = name}

function Student() {
Persion.apply(this,arguments); //apply可以劫持另一个对象大方法
//继承另一个对象的属性
//两个参数obj和args:obj也就是this将代替Person类中的this对象
//args 作为参数传给Function也就是P传给Person
// call 参数列表不同 Function.call(obj,[param1,[,param2,[,…[paramN]]]])
// obj这个对象代替Function类中的this对象 params：一个参数列表

}
const student = new Student(‘lxy’)
student.name //lxy

3.组合式继承（结合以上两种的特点）
function Person(name) {
this.name = name;
}
Person.prototype.getName = function() {
return this.name;
}
function Student() {
// 构造函数继承
Person.apply(this, arguments)
}
// 原型式继承
Student.prototype = new Person();

// 原型的实例等于自身
Student.prototype.constructor = Student;

const student = new Student(‘Back_kk’);

console.log(student.getName()); // Back_kk
4.寄生式组合继承

Student.prototype = Object.create(Persion.protytpe)

Student.prototype.consructor = Student

** 总结：最初使用原型链继承，通过将子类实例的原型指向父类

什么是防抖和节流

本质上是减少回调函数（resize，scroll，mousemove等事件）的调用频率
极大的浪费资源，降低前端性能
为了对这类事件调用次数限制，使用throttle（节流）和debounce（防抖）
减少调用频率

** 定义**：

  • 节流：n秒内只运行一次，若在n秒内重复触发，只生效一次
    **简单写法

  • 防抖：n秒后再执行，若再n秒内被重复调用，则重新计时

代码实现 (节流)

** 使用时间戳写法
function throttled1(fn,delay=500) {

let oldtime = Date.now() 获取调用时时间戳
return function(…args) {
let newtime = Date.now()
if(newtime-oldtime>=delay) {
fn.apply(null,args)
oldtime = Date.now()
}

}
}

**使用定时器写法
inputDom.onInput = function() {
if(t!==null) {
clearTimeout(t);
}
t = setTimeout(()=>{
console.log(this.value)
},500)
}

function throttled12(fn,delay=500) {
let timer = null
return function(…args) {
if(!timer) {
timer = setTimeout(()=>{
fn.apply(this.args)
timer = null
},delay)
}
}
}

区别

函数防抖关注在一定事件连续触发的事件，只在最后执行一次
而节流一段时间只执行一次
函数节流：在一段时间操作中，每一段时间只执行一次，频率较高
的事件中用于提高性能

function findByCut(arr,key) {
let start = 0
let end = arr.length-1
while(start<end) {
let mid = parseInt((start+end)/2);

} else if(key>arr[mid]) {

//arr:数组;key:查找的元素;start:开始索引;end:结束索引

function search2(arr,key,start,end){ 

//首先判断当前起始索引是否大于结束索引,如果大于说明没有找到元素返回-1
if(start > end) {
    return -1;
}
//如果手动调用不写start和end参数会当做第一次运行默认值
//三元表达式:如果不写end参数则为undefined说明第一次调用所以结束索引为arr.length-1
//如果是递归调用则使用传进来的参数end值
var end= end===undefined ? arr.length-1 : end;  

//如果 || 前面的为真则赋值start,如果为假则赋值后面的0
//所以end变量没有写var end = end || arr.length-1;这样如果递归调用时候传参end为0时会被转化为false,导致赋值给arr.length-1造成无限循环溢出;
var start=start || 0;  

//取中间的索引
var mid=parseInt((start+end)/2);  

if(key==arr[mid]){
    //如果找到则直接返回
    return mid;
}else if(key<arr[mid]){  

    //如果key小于则递归调用自身,将结束索引设置为中间索引-1  
    
    return search2(arr,key,start,mid-1);
}else{  

    //如果key大于则递归调用自身,将起始索引设置为中间索引+1
    return search2(arr,key,mid+1,end);
}

}
var arr = [0,13,21,35,46,52,68,77,89,94];
search2(arr, 77); //7
search2(arr, 99); //-1
